PROGRAM -> PROLOG EOL CLASS  
PROLOG -> import "ifj25" for Ifj
CLASS -> class Program { EOL DEF_FUN_LIST } 

DEF_FUN_LIST -> DEF_FUN  DEF_FUN_LIST | ε
DEF_FUN -> static id ( PARAMETER_LIST ) BLOCK EOL | static id BLOCK EOL | static id = ( id ) BLOCK EOL
PARAMETER_LIST -> id PARAMETER_TAIL | ε
PARAMETER_TAIL -> , id PARAMETER_TAIL | ε 
BLOCK -> { EOL STML_LIST } | { ε }
STML_LIST -> STML_LINE  STML_LIST | ε
STML_LINE -> STML EOL
STML -> VAR | ASSIGN | IF | WHILE | CALL | RETURN  
VAR -> var id
ASSIGN -> id = EXPRESSION
IF -> if ( EXPRESSION ) BLOCK else BLOCK
WHILE -> while ( EXPRESSION ) BLOCK
CALL -> id = id ( ARGUMENT_LIST ) | id = GETTER_ID 
ARGUMENT_LIST -> TERM ARGUMENT_TAIL | ε
ARGUMENT_TAIL -> , TERM ARGUMENT_TAIL | ε
TERM -> LITERAL_NUM | LITERAL_STRING | LITERAL_NULL | id | GLOBAL_ID | GETTER_ID
RETURN -> return EXPRESSION
